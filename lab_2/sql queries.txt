CREATE TABLE zxc_music_hall(
	p_id INT,
    city TEXT,
    organization TEXT,
	
    -- ключ
    PRIMARY KEY(p_id)
);

CREATE TABLE zxc_songers(
    -- имя музыканта
    name TEXT,
    
    -- песни делятся на основные которые постоянно исполняются
    -- и которые меняются
    song TEXT,
    
    -- время когда музыкант выступает
    time TEXT,
    
    -- будет ли выступать музыкант в данное время
    is_checked BOOLEAN,
    
    -- внешний ключ
    p_id INT, 
    FOREIGN KEY(p_id) 
    REFERENCES zxc_music_hall(p_id) 
    ON DELETE CASCADE,
    
    -- ключ
    PRIMARY KEY(name, song)
);
